### 1. 订单创建时地址为空的处理逻辑

在`Order.vue`页面中，修改`createOrder`方法：

```javascript
// 创建订单
const createOrder = async () => {
  if (!orderForm.addressBookId) {
    // 跳转到地址管理页面
    router.push({
      path: '/address',
      query: { returnTo: '/order' }
    });
    return;
  }

  try {
    const response = await orderApi.createOrder(orderForm);
    if (response.code === 1) {
      ElMessage.success('订单创建成功');
      // 清空购物车
      localStorage.removeItem('cart');
      // 跳转到订单列表页
      window.location.href = '/order';
    } else {
      ElMessage.error('订单创建失败: ' + response.message);
    }
  } catch (error) {
    ElMessage.error('后端订单创建失败: ' + error.message);
    console.error('Failed to create order:', error);
  }
};
```

### 2. 地址管理页面返回逻辑

```javascript
// 保存地址
const saveAddress = async () => {
  try {
    await formRef.value.validate();
    let response;
    if (form.id) {
      response = await addressApi.updateAddress(form);
    } else {
      response = await addressApi.addAddress(form);
    }
    if (response.code === 1) {
      ElMessage.success(form.id ? '地址更新成功' : '地址添加成功');
      dialogVisible.value = false;
      loadAddressList();
      
      // 检查是否需要返回订单页面
      const returnTo = router.currentRoute.value.query.returnTo;
      if (returnTo) {
        router.push(returnTo);
      }
    } else {
      ElMessage.error(response.message);
    }
  } catch (error) {
    ElMessage.error('保存地址失败');
    console.error('Failed to save address:', error);
  }
};
```

### 3. 订单页面加载地址逻辑

在`Order.vue`页面的`onMounted`钩子中，添加加载默认地址的逻辑：

```javascript
onMounted(() => {
  // 加载地址列表
  loadAddressList();
  // 从本地存储获取购物车数据
  const cartItems = getCartItems();
  // 计算订单总金额
  orderForm.amount = calculateTotalAmount(cartItems);
  // 生成订单号
  orderForm.number = generateOrderNumber();
  // 设置订单时间
  orderForm.orderTime = new Date().toLocaleString();
});

// 加载地址列表
const loadAddressList = async () => {
  try {
    const response = await addressApi.getAddressList();
    if (response.code === 1) {
      addressList.value = response.data;
      // 自动选择默认地址
      const defaultAddress = addressList.value.find(item => item.isDefault === 1);
      if (defaultAddress) {
        selectedAddressId.value = defaultAddress.id;
        orderForm.addressBookId = defaultAddress.id;
      }
    } else {
      ElMessage.error(response.message);
    }
  } catch (error) {
    ElMessage.error('加载地址失败');
    console.error('Failed to load address list:', error);
  }
};
```

### 4. 数据库操作

执行以下SQL语句创建address_book表：

```sql
-- 创建地址簿表
CREATE TABLE `address_book` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '主键',
  `user_id` bigint(20) NOT NULL COMMENT '用户id',
  `consignee` varchar(255) NOT NULL COMMENT '收货人',
  `phone` varchar(20) NOT NULL COMMENT '手机号',
  `province` varchar(255) NOT NULL COMMENT '省',
  `city` varchar(255) NOT NULL COMMENT '市',
  `district` varchar(255) NOT NULL COMMENT '区',
  `detail` varchar(255) NOT NULL COMMENT '详细地址',
  `postcode` varchar(20) DEFAULT NULL COMMENT '邮编',
  `is_default` int(11) DEFAULT '0' COMMENT '状态 0:默认 1:非默认',
  `create_time` datetime NOT NULL COMMENT '创建时间',
  `update_time` datetime NOT NULL COMMENT '更新时间',
  `create_user` bigint(20) NOT NULL COMMENT '创建人',
  `update_user` bigint(20) NOT NULL COMMENT '更新人',
  `is_deleted` int(11) DEFAULT '0' COMMENT '逻辑删除',
  PRIMARY KEY (`id`),
  KEY `idx_user_id` (`user_id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_bin COMMENT='地址簿';
```
